func(2)

def func(a: list, b: int, c = "pear") -> dict:
    """
    This is an exmaple for the help document!
    a, b: position parameters
    c: keyword parameter
    """
    data = a[1] * b + 1
    output = {}
    for i in range(data):
        if i > 1:
            output[i] = c
        else:
            output[i] = "apple"
    return output

a = [0, 1, 2]
b = 2
# -----------------------------
print(func(a, b))
# -----------------------------
print(func(a, b, c = "orange"))

class

class Fruit:
    def __init__(self, name):
        self.name = name
    def get_price(self, weight):
        if self.name == "apple":
            return 3 * weight
        elif self.name == "pear":
            return 2 * weight
        else:
            return 1 * weight
    def get_preference(self, score):
        if score < 7:
            print("No, you dislike its taste!")
        else:
            print("Yes, you do like it!")

# inistialize as an instance
apple = Fruit("apple")
# ---------------------------------
# we bought three kilos of apple 
price = apple.get_price(3)
print(price)
apple.get_preference(8)

pear = Fruit("pear")
# ---------------------------------
# we bought three kilos of pear 
price = pear.get_price(3)
print(price)
pear.get_preference(7)

orange = Fruit("orange")
# ---------------------------------
# we bought three kilos of orange 
price = orange.get_price(3)
print(price)
orange.get_preference(2)
