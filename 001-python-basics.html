def func(a: list, b: int, c = "pear") -> dict:
    """
    This is an exmaple for the help document!
    a, b: position parameters
    c: keyword parameter
    """
    data = a[1] * b + 1
    output = {}
    for i in range(data):
        if i > 1:
            output[i] = c

class

class Fruit:
    def __init__(self, name):
        self.name = name
    def get_price(self, weight):
        if self.name == "apple":
            return 3 * weight
        elif self.name == "pear":
            return 2 * weight
        else:
            return 1 * weight
    def get_preference(self, score):
        if score < 7:
            print("No, you dislike its taste!")
        else:
            print("Yes, you do like it!")

# inistialize as an instance
apple = Fruit("apple")
# ---------------------------------
# we bought three kilos of apple 
price = apple.get_price(3)
print(price)
apple.get_preference(8)
