# Easy multiprocessing

'''
必须要有__name__ == "main" 函数，不然持续程序持续运行
imap_unordered为无序处理，但可以记录进度
'\r'可以保证在一行输出
'''

import sys
import multiprocessing as mp

def f(x):
    return x * x


def main():
    cores = mp.cpu_count()
    pool = mp.Pool(processes=cores)
    xs = range(15000)
    # method 1: map
    print(pool.map(f, xs))  # prints [0, 1, 4, 9, 16]

    # method 2: imap
    for y in pool.imap(f, xs):
        print(y)           # 0, 1, 4, 9, 16, respectively

    # method 3: imap_unordered
    for y in pool.imap_unordered(f, xs):
        print(y)           # may be in any order

    cnt = 0
    for _ in pool.imap_unordered(f, xs):
        # print(_)
        sys.stdout.write('done %d/%d\r' % (cnt, len(xs)))
        cnt += 1

if __name__ == "__main__":
    main()

# Multi-processing with multiple arguments
#!/usr/bin/env python3
import multiprocessing as mp
from functools import partial
from itertools import repeat

def func(a, b):
    return a + b

def main():
    a_args = [1,2,3]
    second_arg = 1
    with mp.Pool(mp.cpu_count()) as pool:
        L = pool.starmap(func, [(1, 1), (2, 1), (3, 1)])
        M = pool.starmap(func, zip(a_args, repeat(second_arg)))
        N = pool.map(partial(func, b=second_arg), a_args)
        assert L == M == N

if __name__=="__main__":
    mp.freeze_support()
    main()
    print("done")

# Shared and return value

import multiprocessing as mp
import time

def func(num, i, output):
    res = func2(num, i)
    output.put(res)

def func2(num, i):
    num.value += i  #子进程改变数值的值，主进程跟着改变
    print(i, num.value)
    return num.value
    # output.put(num.value)

if  __name__=="__main__":

    output = mp.Queue()

    num = mp.Value("d", 10.0) # d表示数值,主进程与子进程共享这个value。（主进程与子进程都是用的同一个value）
    print(num.value)

    runed = [3, 5, 8]
    params = [i for i in range(10)]
    processes = [mp.Process(target=func, args=(num, param, output)) for param in params if param not in runed]

    # Run processes
    for p in processes:
        p.start()
        time.sleep(0.01)

    # Exit the completed processes
    for p in processes:
        p.join()

    # Get process results from the output queue
    results = [output.get() for p in processes]

    print(results)
